	/*FIFASTX-5666  - Error when not entering a card*/
	function handleErrorIcon() {
		const header = document.querySelector(".stx-lt-paymentSelection-header-title");
		if (!header) return;

		const svg = header.querySelector("svg");
		if (!svg) return;

		if (!svg.classList.contains("visa-error-icon")) {
			svg.classList.add("visa-error-icon");
		}

		const isError = svg.classList.contains("tw-text-[var(--messageErrorTitleColor,_#bc1c00)]");
		const isSuccess = svg.classList.contains("tw-text-[var(--messageSuccessTitleColor,_#257a56)]");

		let msg = header.querySelector(".visa-error-text");

		if (isError) {
			if (!msg) {
				msg = document.createElement("span");
				msg.className = "visa-error-text";
				msg.style.color = "#bc1c00";
				msg.style.marginLeft = "8px";

				if (document.querySelector("html[lang=en]")) {
					msg.textContent = "You must use a valid payment method to complete your entry.";
				}
				if (document.querySelector("html[lang=es]")) {
					msg.textContent = "Debes usar un método de pago válido para completar tu inscripción.";
				}
				if (document.querySelector("html[lang=fr]")) {
					msg.textContent = "Vous devez utiliser un moyen de paiement valide pour finaliser votre inscription.";
				}
				if (document.querySelector("html[lang=de]")) {
					msg.textContent = "Sie müssen eine gültige Zahlungsmethode verwenden, um Ihre Anmeldung abzuschließen.";
				}

				header.appendChild(msg);
			} else {
				msg.style.display = "inline";
			}
		} else if (isSuccess && msg) {
			msg.style.display = "none";
		}
	}

	function updateLocalityLabels() {
		const translations = {
			state: {
				en: "Province/State",
				fr: "Province/État",
				es: "Provincia/Estado",
				de: "Provinz/Bundesland"
			},

				fr: "Région",
				es: "Región",
				de: "Region"
			},

				fr: "Département",

		const stateLabel = document.querySelector(

		);
		if (stateLabel && translations.state[htmlLang]) {
			stateLabel.childNodes[0].nodeValue = translations.state[htmlLang] + " ";
		}

		if (regionLabel && translations.region[htmlLang]) {
			regionLabel.childNodes[0].nodeValue = translations.region[htmlLang] + " ";
		}

		const deptLabel = document.querySelector('#field_locality_criteria label[for="locality_DEPT"]');
		if (deptLabel && translations.dept[htmlLang]) {
			deptLabel.childNodes[0].nodeValue = translations.dept[htmlLang] + " ";
		}
	}

	/*FIFASTX-5788  - Hiding the title before the name on the homepage*/
	function hideTitle() {
		let homeTitle = document.querySelector('h3.title4');
		if (homeTitle) {
			homeTitle.childNodes.forEach(node => {
				if (node.nodeType === Node.TEXT_NODE) {
					node.remove();
				}
			});
		}
	}

	/*FIFASTX-5796  - Link wrapping the entire terms and conditions line*/
	function fixTermsAndConditions() {
		const wrapper = document.querySelector(
			".stx-confirmation-terms-and-conditions .tw-mx-2 > a[href='General terms and conditions']"
		);

		if (!wrapper) return;

		const parent = wrapper.parentNode;
		while (wrapper.firstChild) {
			parent.insertBefore(wrapper.firstChild, wrapper);
		}
		parent.removeChild(wrapper);
	}

	function applyTargetBlankToDialogLinks() {
		try {
			if (!location.pathname.startsWith('/secured/content')) return;
			const dialog = document.querySelector('#stx-lt-subscriptionSummary-panel-dialog');
			if (!dialog) {
				console.log("applyTargetBlankToDialogLinks: Dialog not found yet, retrying...");
				setTimeout(applyTargetBlankToDialogLinks, 500);
				return;
			}

			const links = dialog.querySelectorAll('a');
			if (!links.length) {
				console.log("applyTargetBlankToDialogLinks: No links found in dialog.");
				return;
			}

			links.forEach(link => {
				if (!link.hasAttribute('target')) {
					link.setAttribute('target', '_blank');
					console.log("applyTargetBlankToDialogLinks: Added target=_blank to", link.href);
				}
				if (!link.hasAttribute('rel')) {
					link.setAttribute('rel', 'noopener noreferrer');
				}
			});
			console.log("applyTargetBlankToDialogLinks: All dialog links updated.");
		} catch (err) {
			console.error("applyTargetBlankToDialogLinks: Error", err);
		}
	}

		function processContainer(container) {
			if (container.querySelector(".sold_out_cta")) return;

			const availabilityEl = container.querySelector(
				".availability_status .text span[aria-hidden='true']"
			);
			if (!availabilityEl) return;

				"non disponible",
				"nicht verfügbar",
				"no disponible"
			];

			if (!isSoldOut) return;

			const buttonsArea = container.querySelector(".buttons");
			if (!buttonsArea) return;

			btn.style.color = "#FFF";
			btn.style.textAlign = "center";
			btn.style.borderRadius = "1000px";
			btn.style.fontWeight = "600";
			btn.style.textDecoration = "none";

			buttonsArea.appendChild(btn);
		}

		document.querySelectorAll(".availability_container").forEach(processContainer);

		const observer = new MutationObserver((mutations) => {
			mutations.forEach((mutation) => {
				mutation.addedNodes.forEach((node) => {
					if (node.nodeType !== 1) return;

					if (node.matches && node.matches(".availability_container")) {
						processContainer(node);
					}

					const nestedContainers = node.querySelectorAll?.(".availability_container");
					if (nestedContainers) nestedContainers.forEach(processContainer);
				});
			});
		});

		observer.observe(document.body, { childList: true, subtree: true });
	}

			if (existingWrap) existingWrap.remove();
			return;

		if (existingWrap) {
			const master = document.getElementById("seatCategorySelectAll");
			if (!master) return;
	
			const checkedCount = boxes.filter(box =>
				box.classList.contains("p-checkbox-checked")
			).length;
	
			master.checked = checkedCount === boxes.length;
			master.indeterminate = checkedCount > 0 && checkedCount < boxes.length;

function removeCanceledTickets() {
	try {
		const pageWrapper = document.getElementById('page_account_tickets');
		if (!pageWrapper) return;

		const productContainers = pageWrapper.querySelectorAll('.product_container');

		productContainers.forEach(product => {
			const ticketContainers = product.querySelectorAll('.ticket_container');

			ticketContainers.forEach(container => {
				const listItems = container.querySelectorAll('li');

				listItems.forEach(li => {
					const isCanceled = li.querySelector('.content.canceled');
					const hasPrintOptions = li.querySelector('.print_options');
					const isResale = li.querySelector('.resale_info');

					if (isCanceled && hasPrintOptions && !isResale) {
						li.remove();
						
						const seatCategory = container.closest('.seat_category_container');
						if (seatCategory) {
							const summary = seatCategory.querySelector('.tickets_summary');
							if (summary) {
								summary.style.display = 'none';
							}
						}
					}
				});
			});

			const remainingTickets = product.querySelectorAll('.ticket_container li');
			if (remainingTickets.length === 0) {
				product.remove();
			}
		});
	} catch (err) {
		console.error('removeCanceledTickets error:', err);
	}
}
